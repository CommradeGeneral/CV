\newcommand{\daydifftodayi}[3]{%
\setcounter{datefour}{0}%
\myyears = \@dtm@currentyear%
\myprevyear = \@dtm@currentyear%
\advance \myprevyear by -1%
\advance \myyears by -#1%
\setmydatenumber{dateone}{\@dtm@currentyear}{\@dtm@currentmonth}{\@dtm@currentday}%
\setmydatenumber{datetwo}{\@dtm@currentyear}{#2}{#3}%
\setmydatenumber{datethree}{\the\myprevyear}{#2}{#3}%
\addtocounter{dateone}{-\thedatetwo}%
\addtocounter{datefour}{\thedatetwo}%
\addtocounter{datefour}{-\thedatethree}%
\ifnum \value{dateone}<0
	\advance \myyears by -1%
	\addtocounter{dateone}{\thedatefour}%
\fi%
\the\myyears\ years old%
\ifnum \value{dateone}=0 \relax%
\else \ifnum \value{dateone}=1%
	\ and \thedateone\ day%
\else%
	\ and \thedateone\ days%
\fi\fi%
}
